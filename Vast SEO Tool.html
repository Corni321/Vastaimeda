    <!-- JavaScript for Functionality -->

    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const urlInput = document.getElementById('urlInput');

            const analyzeButton = document.getElementById('analyzeButton');

            const buttonText = document.getElementById('buttonText');

            const loadingDots = document.getElementById('loadingDots');

            const reportSection = document.getElementById('reportSection');

            const coreReport = document.getElementById('coreReport');

            const keywordReport = document.getElementById('keywordReport');

            const performanceReport = document.getElementById('performanceReport');

            const noUrlMessage = document.getElementById('noUrlMessage');

            const errorModal = document.getElementById('errorModal');

            const errorMessage = document.getElementById('errorMessage');

            const keywordChartCanvas = document.getElementById('keywordChart');


            let keywordChart = null;


            analyzeButton.addEventListener('click', async () => {

                const url = urlInput.value.trim();

                if (!url) {

                    noUrlMessage.classList.remove('hidden');

                    return;

                }

                noUrlMessage.classList.add('hidden');

                

                // Show loading state

                buttonText.textContent = 'Analyzing';

                loadingDots.classList.remove('hidden');

                analyzeButton.disabled = true;


                try {

                    const response = await fetch(`https://vastaimedia-seo-api.netlify.app/.netlify/functions/analyze?url=${encodeURIComponent(url)}`);

                    if (!response.ok) {

                        throw new Error(`HTTP error! Status: ${response.status}`);

                    }

                    const data = await response.json();

                    displayReport(data);

                } catch (error) {

                    showErrorModal(`Failed to analyze URL: ${error.message}. Please check the URL and try again.`);

                } finally {

                    // Reset button state

                    buttonText.textContent = 'Analyze';

                    loadingDots.classList.add('hidden');

                    analyzeButton.disabled = false;

                }

            });


            const displayReport = (data) => {

                reportSection.classList.remove('hidden');

                

                // Core SEO

                coreReport.innerHTML = `

                    <p class="mb-2"><span class="font-bold">Title:</span> ${data.title || 'N/A'}</p>

                    <p class="mb-2"><span class="font-bold">Meta Description:</span> ${data.metaDescription || 'N/A'}</p>

                    <p class="mb-2"><span class="font-bold">H1 Tag:</span> ${data.h1 || 'N/A'}</p>

                    <div class="mt-4">

                        <p class="font-bold mb-2">Internal & External Links:</p>

                        <p class="text-sm text-gray-600">Total Links: ${data.linkCount.total}</p>

                        <p class="text-sm text-gray-600">Internal Links: ${data.linkCount.internal}</p>

                        <p class="text-sm text-gray-600">External Links: ${data.linkCount.external}</p>

                    </div>

                `;


                // Keyword Density

                const sortedKeywords = Object.entries(data.keywordDensity)

                    .sort(([, a], [, b]) => b - a)

                    .slice(0, 10);

                

                const keywordLabels = sortedKeywords.map(item => item[0]);

                const keywordData = sortedKeywords.map(item => item[1]);


                if (keywordChart) {

                    keywordChart.destroy();

                }


                keywordChart = new Chart(keywordChartCanvas, {

                    type: 'bar',

                    data: {

                        labels: keywordLabels,

                        datasets: [{

                            label: 'Keyword Frequency (%)',

                            data: keywordData,

                            backgroundColor: 'rgba(59, 130, 246, 0.6)',

                            borderColor: 'rgba(59, 130, 246, 1)',

                            borderWidth: 1

                        }]

                    },

                    options: {

                        responsive: true,

                        maintainAspectRatio: false,

                        scales: {

                            y: {

                                beginAtZero: true,

                                title: {

                                    display: true,

                                    text: 'Frequency (%)'

                                }

                            }

                        }

                    }

                });


                // Performance Report

                performanceReport.innerHTML = `

                    <div class="flex flex-col md:flex-row items-center space-y-4 md:space-y-0 md:space-x-4">

                        <div class="flex-1 w-full text-center">

                            <h4 class="font-bold mb-2">Page Speed</h4>

                            <div class="progress-bar-container">

                                <div class="progress-bar bg-green-500" style="width: ${data.pageSpeed ? Math.min(data.pageSpeed, 100) : 0}%;">

                                    <span class="progress-bar-label">${data.pageSpeed ? data.pageSpeed + '%' : 'N/A'}</span>

                                </div>

                            </div>

                            <p class="text-sm text-gray-600 mt-2">A higher percentage is better.</p>

                        </div>

                        <div class="flex-1 w-full text-center">

                            <h4 class="font-bold mb-2">Mobile Friendliness</h4>

                            <p class="text-lg font-bold ${data.mobileFriendly ? 'text-green-500' : 'text-red-500'}">${data.mobileFriendly ? 'Mobile Friendly' : 'Not Mobile Friendly'}</p>

                            <p class="text-sm text-gray-600">Based on viewport meta tag.</p>

                        </div>

                    </div>

                `;

            };


            const showErrorModal = (message) => {

                errorMessage.textContent = message;

                errorModal.style.display = 'block';

            };


            window.onclick = function(event) {

                if (event.target == errorModal) {

                    errorModal.style.display = "none";

                }

            };

        });

    </script>
